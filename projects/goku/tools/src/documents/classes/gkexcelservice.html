toDualityArray

  • toDualityArray(titles: GKTableTitle[], data: TableData[]): string[][]
    • Defined in services/GKExcel/gkexcel.service.ts:75

    将数据格式转换成适用于excel的二元数组结构

    Parameters

    • titles: GKTableTitle[]

      表头列表

    • data: TableData[]

      表格数据本身

      示例:

       
      export class DemoService {
 constructor(private gkXlsx: GKExcelService){}

 titles: GKTableTitle[] = [
   { text: '姓名', prop: 'name' },
   { text: '年龄', prop: 'age' }
 ]

 data = [
   { name: '小明', age: 18 },
   { name: '李磊', age: 19 }
 ]

 formater(){
   let res = this.gkXlsx.toDualityArray(this.titles, this.data);
   console.log(res);
   // [ ['姓名',年龄], ['小明',18], ['李磊',19] ]
 }
}

    Returns string[][]
